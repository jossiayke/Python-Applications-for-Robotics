 50def pickup_tray(robot: str, tray: str, table: str):
 51    """
 52    `robot` picks up `tray` from `table`
 53
 54    Note: This function can only be executed if `robot` is 'ceiling_robot'
 55
 56    Preconditions:
 57        - `robot` is 'robot_ceiling'
 58        - `tray` is located on `table`
 59        - `robot`'s gripper is empty
 60
 61    Effects:
 62        - `robot`'s gripper is not empty
 63        - `robot`'s gripper is holding the tray
 64        - location of `tray` is not on `table`
 65        - location of `tray` is in `robot`'s gripper
 66
 67    Args:
 68        robot (str): The robot used to perform this action
 69        tray (str): The tray to pick up
 70            ('w' - 'yellow_tray or 'm' - 'gray_tray')
 71        table (str): Table where the tray is located
 72            ('wood_tray' or 'metal_tray')
 73    """
 74
 75    robot_gripper = envt_state[robot]['empty']
 76    item_tray = ''
 77    if tray == 'w':
 78        item_tray = 'wood_tray'
 79        tray_check = (envt_state['table'][item_tray+'_table'] and
 80                      envt_state['kitting_tray']
 81                      ['yellow_tray']['on_table'])
 82
 83    else:
 84        item_tray = 'metal_tray'
 85
 86        tray_check = (envt_state['table'][item_tray+'_table'] and
 87                      envt_state['kitting_tray']['gray_tray']['on_table'])
 88
 89    if(robot == 'robot_ceiling' and tray_check and robot_gripper):
 90        envt_state[robot]['empty'] = False
 91        envt_state[robot]['hold_tray'] = True
 92
 93        if tray == 'w':
 94            envt_state['table']['wood_tray_table'] = False
 95            envt_state['kitting_tray']['yellow_tray']['on_table'] = False
 96            envt_state['kitting_tray']['yellow_tray']['on_robot'] = True
 97
 98        else:
 99            envt_state['table']['metal_tray_table'] = False
100            envt_state['kitting_tray']['gray_tray']['on_table'] = False
101            envt_state['kitting_tray']['gray_tray']['on_robot'] = True
102
103        envt_state[robot]['hold_tray'] = True
104
105    print(f"pickup_tray({robot}, {item_tray}, {table})")

108def place_tray(robot: str, tray: str, agv: str):
109    """
110    `robot` places `tray` on `agv`.
111
112    Note: This function can only be executed if `robot` is 'robot_ceiling'.
113
114    Preconditions:
115        - `robot` is 'robot_ceiling'
116        - `robot`'s gripper is holding `tray`
117        - `tray` is located in `robot`'s gripper
118        - location of `tray` is in `robot`'s gripper
119
120    Effects:
121        - `robot`'s gripper is empty
122        - `robot`'s gripper is not holding `tray`
123        - location of `tray` is not in `robot`'s gripper
124        - location of `tray` is `agv`
125
126    Args:
127        robot (str): The robot used to perform this action
128        tray (str): The tray to place on `agv`
129            ('w' - 'yellow_tray or 'm' - 'gray_tray')
130        agv (str): AGV where `tray` is placed
131            ('agv1', 'agv2', 'agv3', or 'agv4')
132    """
133
134    robot_gripper = (not envt_state[robot]['empty'] and
135                     envt_state[robot]['hold_tray'])
136    if tray == 'w':
137        tray = 'wood_tray'
138        tray_check = (not envt_state[robot]['empty'] and
139                      envt_state['kitting_tray']
140                      ['yellow_tray']['on_robot'])
141
142    elif tray == 'm':
143        tray = 'metal_tray'
144        tray_check = (not envt_state[robot]['empty'] and
145                      envt_state['kitting_tray']['gray_tray']['on_robot'])
146
147    if(robot == 'robot_ceiling' and tray_check and robot_gripper):
148        envt_state[robot]['empty'] = True
149        envt_state[robot]['hold_tray'] = False
150        if tray == 'wood_tray':
151            envt_state['kitting_tray']['yellow_tray']['on_robot'] = False
152            envt_state['kitting_tray']['yellow_tray']['on_agv'] = True
153        else:
154            envt_state['kitting_tray']['gray_tray']['on_robot'] = False
155            envt_state['kitting_tray']['gray_tray']['on_agv'] = True
156
157    # agv is an integer between 1 - 4
158
159    envt_state['agv'][agv] = True
160
161    print(f"place_tray({robot}, {tray}, {agv})")

164def pickup_part(robot: str, bin: str, part: str):
165    """
166    `robot` picks up `part` from `bin`.
167
168    Preconditions:
169        - `robot`'s gripper is empty
170        - `bin` has parts of type `part`
171        - The number of parts in `bin` is > 0
172
173    Effects:
174        - Decrease the number of parts in `bin` by 1
175
176    Args:
177        robot (str): Robot used to perform this action
178            ('robot_ceiling' or 'robot_floor')
179        bin (str): Bin from which a part is removed
180            (1 - 'bin1', ..., or 4 - 'bin4')
181        part (str): The part to pick up
182            ('red_part', 'green_part', or 'blue_part')
183    """
184
185    robot_gripper = envt_state[robot]['empty']
186
187    part_no = envt_state[part]
188
189    if (robot_gripper and (bin in '1234') and part_no > 0):
190
191        envt_state[part] -= 1
192
193    print(f"pickup_part({robot}, {bin}, {part})")

196def place_part(robot: str, tray: str, part: str, agv: str):
197    """
198    `robot` places `part` in `tray`, which is located on `agv`
199
200    Preconditions:
201        - `robot`'s gripper is holding `part`
202        -  location of `tray` is `agv`
203        - `tray` is not complete
204
205    Effects:
206        - `tray` has `part`
207        - `robot`'s gripper is not holding `part`
208
209    Args:
210        robot (str): Robot used to perform this action
211            ('robot_ceiling' or 'robot_floor')
212        tray (str): Tray in which `part` is placed
213            ('w' - 'yellow_tray or 'm' - 'gray_tray')
214        part (str): The part to place
215            ('red_part', 'green_part', or 'blue_part')
216        agv (str): AGV where `tray` is located.
217            ('agv1', 'agv2', 'agv3', or 'agv4')
218    """
219
220    robot_gripper = (not envt_state[robot]['empty'] and
221                     envt_state[robot]['hold_part'])
222    if tray == 'w':
223        tray = 'wood_tray'
224        tray_check = (not envt_state[robot]['empty'] and
225                      envt_state['kitting_tray']['yellow_tray']['on_agv'])
226
227    elif tray == 'm':
228        tray = 'metal_tray'
229        tray_check = (not envt_state[robot]['empty'] and
230                      envt_state['kitting_tray']['gray_tray']['on_agv'])
231
232    comp = envt_state['tray_status']
233
234    if(robot_gripper and tray_check and comp):
235
236        if part == 'red_part':
237            envt_state['parts_In_T']['red'] += 1
238        elif part == 'green_part':
239            envt_state['parts_In_T']['green'] += 1
240        else:
241            envt_state['part_in_T']['blue'] += 1
242
243        envt_state[robot]['hold_part'] = False
244        envt_state[robot]['empty'] = True
245
246    print(f"place_part({robot}, {tray}, {part}, {agv})")

249def ship_agv(agv: str, tray: str, station: str):
250    """
251    Ship `agv` to assembly station `station` if `tray` is complete
252
253    Preconditions:
254        - `tray` is complete (it has all the parts needed)
255        - `tray` is on `agv`
256
257    Effects:
258        - `agv` location is `station`
259
260    Args:
261        agv (str): AGV to ship
262            ('agv1', 'agv2', 'agv3', or 'agv4')
263        tray (str): Tray located on `agv`
264            ('yellow_tray' or 'gray_tray')
265        station (str): Assembly station where to ship `agv`
266            ('as1', 'as2', 'as3', or 'as4')
267    """
268    if tray == 'yellow_tray':
269        tray_check = envt_state['kitting_tray'][tray]['on_agv']
270
271    elif tray == 'gray_tray':
272        tray_check = envt_state['kitting_tray'][tray]['on_agv']
273
274    stat = envt_state['tray_status']
275
276    if stat and tray_check:
277        print(f"ship_agv({agv}, {station})")
278    else:
279        print("NO SOLUTION FOUND - Assembly station not known!")

282def initial_state(part):
283    """
284    Ask user question about desired part.
285
286    Preconditions:
287        - no preconditions
288
289    Effects:
290        - Record number of parts in the bins
291
292    Args:
293        part (str): type of part to be placed on bin
294                    (red_part), (green_part), and (blue_part)
295    """
296
297    while True:
298
299        parts_in = input(f"How many {part}_parts in the workcell [0, 4, 9]? ")
300        parts_in.replace(" ", "")
301        if parts_in not in '0123456789':
302            print("Non-integer entered. Please Re-enter a number")
303            continue
304        elif parts_in not in '049':
305            print("Wrong number entered. Choose a number from list")
306            continue
307        envt_state[part+'_part'] = int(parts_in)
308
309        if parts_in != '0':
310
311            used = part
312            while True:
313                parts_bin = input(
314                    f"In which bin are {part}_parts located [1, 2, 3, 4]? ")
315                parts_bin.replace(" ", "")
316                if parts_bin not in '0123456789':
317                    print("Non-integer entered. Please Re-enter a number")
318                    continue
319                if parts_bin not in '1234':
320                    print("Wrong number entered. Choose a number from list")
321                    continue
322                envt_state[part+'_location']['bin' + parts_bin] = True
323                break
324        else:
325            used = ''
326        break
327    return used
